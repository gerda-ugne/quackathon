	/**
	 * Places enemies on the map randomly.
	 * 
	 */
	public void addEnemies()
	{
		int enemyCount = 3;
		
		
		Random rand = new Random();
		int enemyPositionX = rand.nextInt(MAP_SIZE);
		int enemyPositionY = rand.nextInt(MAP_SIZE);
		
		for(int i=0; i<enemyCount; i++)
		{
			while(!((map[enemyPositionX][enemyPositionY]).getCharacter() == '.') || map[enemyPositionX][enemyPositionY].getCharacter() == 1 || map[enemyPositionX][enemyPositionY].getCharacter() == 2)
			{
				enemyPositionX = rand.nextInt(MAP_SIZE);
				enemyPositionY = rand.nextInt(MAP_SIZE);
			}
			
			map[enemyPositionX][enemyPositionY].setCharacter('H');	
			enemyPositionX = rand.nextInt(MAP_SIZE);
			enemyPositionY = rand.nextInt(MAP_SIZE);
			
		}
	}
	
	public void initialiseMap()
	{
		generateMap();
		addEnemies();
	}
